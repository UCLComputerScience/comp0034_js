Open a JavaScript console in the developer tools for your browser.

Copy and paste sections of the JavaScript following JavaScript code and paste into the console. Enter to see the results.

/* SELECTING ELEMENTS */

// 1. Get all elements using a class
var content = document.querySelectorAll(".lead p");
console.log(content[0]);
console.log(content[1]);

// 2. Get a single element using its id
var myTitleLink = document.querySelector("#aLink");
console.log("This is a node of type: ", myTitleLink.nodeType);
console.log("Inner HTML: ", myTitleLink.innerHTML);

// 3. Get elements using the HTML tag name
var myParas = document.querySelectorAll("p");
console.log("Paragraphs: ", myParas.length);


/* CHANGING AND ADDING ELEMENTS: Scroll back to the top of the page after executing each section of code to see the changes */

// 1. Change alignment of the main section
var mainContent = document.querySelector("#mainContent");
mainContent.setAttribute("align","center");

// 2. Change content of the first h1 element in the mainContent section
var arrayOfH1s = mainContent.querySelectorAll("h1");
arrayOfH1s[0].innerHTML = "This is a new title";

//3. Add a new heading to the end of the 'mainContent' div
var newHeading = document.createElement("h1");
newHeading.innerHTML = "I am a new heading!";
mainContent.appendChild(newHeading);
